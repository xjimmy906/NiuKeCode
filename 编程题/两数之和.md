function twoSum( $numbers ,  $target )
{
    // write code here
    $len = count($numbers)-1;
    for($i=0;$i<=$len;$i++){
        for($j=$i+1;$j<=$len;$j++){
            if($numbers[$i]+$numbers[$j] == $target){
                return[$i+1,$j+1];
            }
        }
    }
    return false;
}
